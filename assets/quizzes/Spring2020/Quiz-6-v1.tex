\problem{8} Let $y=3\sqrt[3]{x}.$
	\begin{subproblems}
	\item Find the linearization, $L(x)$, of $y=3\sqrt[3]{x}$ at $x=1000.$
	\vfill
	\item Use the linearization from part (b) to estimate $\sqrt[3]{999}.$ Write your answer in decimal form.
	\vspace{.8in}
	\end{subproblems}
	
\problem{9} The half-life of Strontium-90 is 28 days. 
	\begin{subproblems}
	\item Suppose a sample has an initial mass of 30 mg. Find a formula for the mass remaining after $t$ days.
	\vfill
	\item How long would it take for the sample to decay to a mass of 2 mg?
	\vfill
	\item Sketch the graph of the mass function.
	\vspace{1.5in}
	\end{subproblems}
